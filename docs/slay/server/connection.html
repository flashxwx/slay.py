161    def __loop_for_reopen(self, reopen_interval: int):
162        while self.__reopen_attempts != 0:
163
164            self.log_adapter.info(
165                f"Trying to reopen in {reopen_interval} seconds"
166            )
167
168            time.sleep(reopen_interval)
169
170            if self.__is_dont_reopen_code:
171                break
172            
173            self.status = 0
174            self.open()

177                break

179            if self.__reopen_attempts < 0:
180                continue

182            self.__reopen_attempts -= 1

185                return

189    def open(self, new_thread: bool = False):
190        with self.status_lock:
191            if self.status == 1:
192                self.log_adapter.warning(
193                    "Cannot open a connection that is being opened."
194                )
195                return
196            if self.status == 2:
197                self.log_adapter.warning(
198                    "Cannot open a connection that is already opened."
199                )
200                return
201
202            self.status = 1
203
204        self.websocket = WebSocketApp(
205            f"wss://{self.socket.ip_addr}:{self.socket.port}",
206            on_open=self.__on_open,
207            on_message=self.__on_message,
208            on_error=self.__on_error,
209            on_close=self.__on_close,
210        )
211
212        run_forever_kwargs = {
213            "sockopt": (
214                (socket.SOL_SOCKET, socket.SO_KEEPALIVE, 1),
215                (socket.IPPROTO_TCP, socket.TCP_KEEPIDLE, 5),
216                (socket.IPPROTO_TCP, socket.TCP_KEEPINTVL, 2),
217                (socket.IPPROTO_TCP, socket.TCP_KEEPCNT, 3),
218            ),
219            "sslopt": {"context": ssl_context}
220        }
221
222        if new_thread:
223            Thread(
224                target=self.websocket.run_forever,
225                kwargs=run_forever_kwargs
226            ).start()
227
228            return

230        self.websocket.run_forever(**run_forever_kwargs)

322    def __on_close(self, websocket: WebSocketApp, code: int, message: str):
323        error = self.websocket_error
324
325        if hasattr(self.websocket_error, "args"):
326            if len(error.args) == 0:
327                self.websocket_error.args = (None, None)
328
329            elif error.args[0] == "Connection to remote host was lost.":
330                self.websocket_error.args = (-1,) + error.args
331            else:
332                self.websocket_error.args = (-1, str(error))
333        else:
334            self.websocket_error.args = (-1, str(error))
335
336        code, message = self.websocket_error.args
337
338        self.status = 0
339        self.__close_event.set()
340
341        self.log_adapter.info("Connection has been closed.")
342
343        if code in websocket_dont_reopen_codes:
344            self.__is_dont_reopen_code = True
345
346        self.__trigger_event_callback("on_close", code, message)

348    def __trigger_event_callback(
349        self, event_name: str, *args: any, **kwargs: any
350    ):
351        event_callback = self.event_callback_dict.get(event_name)
352
353        if not event_callback:
354            return
355
356        if isinstance(event_callback, Callable):
357            event_callback(self, *args, **kwargs)
358            return
359
360        for callback in event_callback:
361            callback(self, *args, **kwargs)
