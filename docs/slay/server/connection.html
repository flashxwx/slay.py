193    def __loop_for_reopen(self, reopen_interval: int):
194        while self.__reopen_attempts != 0:
195
196            if self.__is_dont_reopen_code:
197                break
198
199            if self.internal_logging_switch:
200                self.log_adapter.info(
201                    f"Trying to reopen in {reopen_interval} seconds"
202                )
203
204            time.sleep(reopen_interval)
205
206            if self.__is_dont_reopen_code:
207                break
208            
209            # wait for the end of all sub threads here
210            while True:
211                if self.__running_sub_thread_count == 0:
212                    break
213
214                try:
215                    number = self.__thread_end_signal_channel.get(timeout=self.thread_end_signal_timeout)
216                except:
217                    if self.internal_logging_switch:
218                        self.log_adapter.critical(f"Failed to reopen the connection. There's still a running thread (count: {self.__running_sub_thread_count}) after {self.thread_end_signal_timeout} seconds since the connection was closed.")
219                    return
220
221                self.__running_sub_thread_count -= number
222
223            self.open()
224
225            if self.__is_dont_reopen_code:
226                break
227
228            if self.__reopen_attempts < 0:
229                continue
230
231            self.__reopen_attempts -= 1
232        else:
233            if self.__is_dont_reopen_code:
234                return
235
236            if self.internal_logging_switch:
237                self.log_adapter.critical(str(self.websocket_error.args[1]))

391    def get_game_now_timestamp(self):
392        if self.game_tick == None:
393            return None
394
395        if self.game_tick < 0:
396            game_tick = abs(self.game_tick) + self.max_round_tick
397        else:
398            game_tick = self.game_tick

400        total_seconds = game_tick // 20

404    def __func_for_response_event_timeout(
405        self, event_name_queue: Queue, target_event_name: str,
406        timeout_func: Callable, timeout: float,
407        args: tuple[any] = None, kwargs: dict[str, any] = None
408    ):
409        start_at = time.time()
410
411        while True:
412            try:
413                event_name = event_name_queue.get(timeout=timeout)
414
415                if event_name == target_event_name:
416                    break
417
418                if (time.time() - start_at) > timeout:
419                    raise TimeoutError
420            except:
421                timeout_func(*args if args else (), **kwargs if kwargs else {})
422                break
423
424        self.__event_name_queues.remove(event_name_queue)

448    def __on_message(self, websocket: WebSocketApp, message: str):
449        self.__trigger_event_callback("on_message", message)
450
451        if self.socket == Socket.SOCIAL:
452            event_name, response = parse_social_response_message(message)
453
454            if not event_name:
455                return
456        
457        elif message[:3] == "upd":
458            self.game_tick, splitted_message = in_game_update_info_parser(message)
459
460            for event_name, response in in_game_update_info_generator(splitted_message, self.event_callback_dict):
461                self.__trigger_event_callback(event_name, response)
462
463            if self.enable_replay_cache and self.__can_start_record_replay:
464                self.replay_cache.append(message)
465
466            return
467        else:
468            messageType, _, messageBody = message.partition("$")
469
470            metadata = response_dict.get(messageType)
471
472            if not metadata:
473                return
474
475            event_name = metadata[0]
476
477            for event_name_queue in self.__event_name_queues:
478                event_name_queue.put_nowait(event_name)
479            
480            if self.enable_replay_cache:
481                if messageType == "next-maps":
482                    if len(self.replay_cache) > 1:
483                        self.last_replay_cache = self.replay_cache.copy()
484                        self.replay_cache.clear()
485                        self.replay_cache.append("replay-version=4")
486
487                    self.__can_start_record_replay = True
488                    self.replay_cache.append(message)
489
490                elif self.__can_start_record_replay and (messageType != "pid") and (messageType != "stats") and (messageType != "init"):
491                    self.replay_cache.append(message)
492
493            if event_name == "on_id":
494                self.__reopen_attempts = self.___reopen_attempts
495
496            if event_name == "on_game_init":
497                response: Info.GameInitial = parse_response_body(messageBody, metadata)
498
499                self.max_round_tick = response.game_data.max_round_ticks
500
501            elif event_name == "on_game_stats":
502                response = parse_response_body(messageBody, metadata)
503                if response.exit:
504                    self.__can_start_record_replay = False
505                    self.game_max_tick = None
506                    self.game_tick = None
507
508            elif event_name == "on_server_message":
509                response = parse_response_body(messageBody, metadata)
510                if response.type == "success":
511                    if self.internal_logging_switch:
512                        self.log_adapter.info(response.content)
513                elif response.type == "error":
514                    if self.internal_logging_switch:
515                        self.log_adapter.error(response.content)
516            else:
517                if event_name not in self.event_callback_dict:
518                    return
519
520                response = parse_response_body(messageBody, metadata)
521
522        self.__trigger_event_callback(event_name, response)

524    def __on_error(self, websocket: WebSocketApp, error: WebSocketException):
525        if error.args[0] == "'NoneType' object has no attribute 'sock'":
526            return
527
528        if isinstance(error, TimeoutError):

535        else:

540        if self.internal_logging_switch:

545    def __on_close(self, websocket: WebSocketApp, code: int, message: str):
546        error = self.websocket_error
547  
548        if hasattr(self.websocket_error, "args"):
549            if len(error.args) == 0:
550                self.websocket_error.args = (None, None)
551
552            elif error.args[0] == "Connection to remote host was lost.":
553                self.websocket_error.args = (-1,) + error.args
554            else:
555                self.websocket_error.args = (-1, str(error))
556        else:
557            self.websocket_error.args = (-1, str(error))
558
559        code, message = self.websocket_error.args
560
561        self.status = 0
562        self.__close_event.set()
563
564        if self.internal_logging_switch:
565            self.log_adapter.info(f"Connection has been closed [Code: {code}].")
566
567        if code in websocket_dont_reopen_codes:
568            self.__is_dont_reopen_code = True
569
570        self.__trigger_event_callback("on_close", code, message)
571
572        if self.__on_main:
573            rel.stop()

575    def __trigger_event_callback(
576        self, event_name: str, *args: any, **kwargs: any
577    ):
578        for event_name_queue in self.__event_name_queues:
579            event_name_queue.put_nowait(event_name)
580
581        event_callback = self.event_callback_dict.get(event_name)
582
583        if not event_callback:
584            return
585
586        if isinstance(event_callback, Callable):
587            event_callback(self, *args, **kwargs)
588            return
589
590        for callback in event_callback:
591            callback(self, *args, **kwargs)
